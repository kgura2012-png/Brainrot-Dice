<script>

const allColors=["red","orange","yellow","green","blue","purple"];

let diceCount = 3;

/* COLORS */
function getActiveColors(){
let checks=document.querySelectorAll(".panel input");
let active=[];

checks.forEach(c=>{
if(c.checked) active.push(c.value);
});

if(active.length===0){
checks.forEach(c=>c.checked=true);
return allColors;
}

return active;
}

/* UI */
function updateDiceUI(){
document.getElementById("diceCountText").innerText=diceCount;
renderDice();
}

function changeDice(val){
diceCount+=val;
if(diceCount<1) diceCount=1;
if(diceCount>6) diceCount=6;
updateDiceUI();
}

/* DICE */
function renderDice(){
let area=document.getElementById("diceArea");
area.innerHTML="";

for(let i=0;i<diceCount;i++){
let div=document.createElement("div");
div.className="dice yellow";
area.appendChild(div);
}
}

/* ANIMATION */
function animateDice(type){
let dices=document.querySelectorAll(".dice");

dices.forEach(d=>{
d.classList.remove("shake","spin");
void d.offsetWidth;

if(type==="mega") d.classList.add("spin");
else d.classList.add("shake");
});
}

/* ROLL */
function rollDice(){
let dices=document.querySelectorAll(".dice");
let colors=getActiveColors();

animateDice("roll");

setTimeout(()=>{
dices.forEach(dice=>{
let color=colors[Math.floor(Math.random()*colors.length)];
dice.className="dice "+color;
});

document.getElementById("result").innerHTML="🎲 Кубики брошены!";
},400);
}

/* MEGA */
function megaRoll(){
let dices=document.querySelectorAll(".dice");
let colors=getActiveColors();

animateDice("mega");

setTimeout(()=>{
dices.forEach(dice=>{
let color=colors[Math.floor(Math.random()*colors.length)];
dice.className="dice "+color;
});

document.getElementById("result").innerHTML="⚡️ MEGA ROLL УСПЕШЕН!";
},700);
}

/* PANEL */
document.addEventListener("keydown",function(e){
if(e.key.toLowerCase()==="q"){
document.getElementById("panel").classList.toggle("show");
}
});

/* INIT */
updateDiceUI();
